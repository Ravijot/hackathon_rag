🔹 Python Example

python
import base64
import requests
from bs4 import BeautifulSoup

def embed_images_in_html(html_path, output_path):
    with open(html_path, "r", encoding="utf-8") as f:
        soup = BeautifulSoup(f, "html.parser")

    for img in soup.find_all("img"):
        src = img.get("src")
        if not src:
            continue
        try:
            # Only handle absolute URLs
            if src.startswith("http"):
                r = requests.get(src)
                mime = r.headers.get("Content-Type", "image/png")
                encoded = base64.b64encode(r.content).decode("utf-8")
                img["src"] = f"data:{mime};base64,{encoded}"
        except Exception as e:
            print(f"Could not embed {src}: {e}")

    # Save single HTML file
    with open(output_path, "w", encoding="utf-8") as f:
        f.write(str(soup))

# Usage
embed_images_in_html("page.html", "page_singlefile.html")
